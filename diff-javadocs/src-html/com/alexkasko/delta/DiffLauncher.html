001package com.alexkasko.delta;
002
003import org.apache.commons.cli.*;
004import org.apache.commons.io.IOCase;
005
006import java.io.File;
007
008import static java.lang.System.out;
009
010/**
011 * Delta diff launcher class
012 *
013 * @author alexkasko
014 * Date: 11/18/11
015 */
016public class DiffLauncher {
017    private static final String HELP_OPTION = "help";
018    private static final String OUTPUT_OPTION = "out";
019    private static final String CASE_SENSITIVE_OPTION = "case-sensitive";
020
021    /**
022     * app entry point
023     */
024    public static void main(String[] args) throws Exception {
025       Options options = new Options();
026        try {
027            options.addOption("h", HELP_OPTION, false, "show this page");
028            options.addOption("o", OUTPUT_OPTION, true, "output file path");
029            options.addOption("c", CASE_SENSITIVE_OPTION, true, "case sensitive [y/n]");
030            CommandLine cline = new GnuParser().parse(options, args);
031            String[] argList = cline.getArgs();
032            if (cline.hasOption(HELP_OPTION)) {
033                throw new ParseException("Printing help page:");
034            } else if(2 == argList.length && cline.hasOption(OUTPUT_OPTION)) {
035                final IOCase caseSensitive;
036                if (cline.hasOption(CASE_SENSITIVE_OPTION)) {
037                    String val = cline.getOptionValue(CASE_SENSITIVE_OPTION);
038                    if ("y".equalsIgnoreCase(val)) {
039                        caseSensitive = IOCase.SENSITIVE;
040                    } else if ("n".equalsIgnoreCase(val)) {
041                        caseSensitive = IOCase.INSENSITIVE;
042                    } else {
043                        throw new ParseException("Invalid case arg: [" + val + "], should be [y] or [n]");
044                    }
045                } else caseSensitive = IOCase.SYSTEM;
046                new DirDeltaCreator().create(new File(argList[0]), new File(argList[1]), new File(cline.getOptionValue(OUTPUT_OPTION)), caseSensitive);
047            } else {
048                throw new ParseException("Incorrect arguments received!");
049            }
050        } catch (ParseException e) {
051            HelpFormatter formatter = new HelpFormatter();
052            out.println(e.getMessage());
053            formatter.printHelp("java -jar delta-diff.jar [-c y/n] dir1 dir2 -o out.zip", options);
054        }
055    }
056}
